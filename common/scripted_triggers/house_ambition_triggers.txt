# ==============================================================================

# Is this character a valid, balance-safe partner for scope:hsm ?
# Expects: scope:hsm  = the house member looking to marry
#          scope:cand = this candidate (saved before the check)
ham_valid_spouse_trigger = {
	NOT = { this = scope:hsm }
	is_alive = yes
	is_adult = yes
	is_visibly_fertile = yes
	is_married = no
	is_betrothed = no
	is_imprisoned = no
	NOT = { has_trait = eunuch }
	NOT = { has_trait = devoted }

	# Opposite sex so the union can produce children
	OR = {
		AND = { scope:hsm = { is_male = yes } is_female = yes }
		AND = { scope:hsm = { is_female = yes } is_male = yes }
	}

	# Don't pull into our own bloodline
	NOT = { dynasty = scope:hsm.dynasty }
	NOT = { is_close_family_of = scope:hsm }

	# Only people worth investing in: no children of their own yet
	NOT = { any_child = { always = yes } }

	# Vanilla marriage legality (faith, age, betrothal, etc.)
	could_marry_character_trigger = { CHARACTER = scope:hsm }

	# Vanilla AI will NOT give up a primary heir into another house,
	# so skip anyone who is the primary heir of a title their parent holds.
	NOT = {
		any_parent = {
			any_held_title = {
				exists = primary_heir
				primary_heir = scope:cand
			}
		}
	}
}

# Does this ruler meet the house's minimum-rank setting?
# Expects: scope:hsm = the house member (house holds var ham_min_tier)
ham_rank_ok_trigger = {
	trigger_if = {
		limit = { scope:hsm.house = { var:ham_min_tier >= 6 } }
		is_independent_ruler = yes
		highest_held_title_tier >= tier_empire
	}
	trigger_else = {
		highest_held_title_tier >= scope:hsm.house.var:ham_min_tier
	}
}
