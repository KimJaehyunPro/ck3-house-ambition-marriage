# Mirrors the proven-working backup interaction structure exactly
# (no category, no common_interaction, no localization on send_option).

house_ambition_options = {
	is_shown = { scope:actor = { is_ai = no } }
	auto_accept = { always = yes }

	on_accept = {
		scope:recipient = {
			# TOGGLE: check the box to flip the active state for the whole house.
			if = {
				limit = { scope:expansionist_marry = yes }
				if = {
					limit = { NOT = { any_in_global_list = { variable = ha_expansionist_list this = scope:recipient.house } } }
					add_to_global_variable_list = { name = ha_expansionist_list target = scope:recipient.house }
					scope:recipient.house = {
						add_house_modifier = ha_expansionist_modifier
						if = {
							limit = { NOT = { has_variable = ha_min_tier } }
							set_variable = { name = ha_min_tier value = 2 }
						}
					}
				}
				else = {
					remove_list_global_variable = { name = ha_expansionist_list target = scope:recipient.house }
					scope:recipient.house = { remove_house_modifier = ha_expansionist_modifier }
				}
			}

			# MINIMUM RANK: highest checked box wins.
			scope:recipient.house = {
				if = {
					limit = { scope:rank_hegemon = yes }
					set_variable = { name = ha_min_tier value = 6 }
				}
				else_if = {
					limit = { scope:rank_empire = yes }
					set_variable = { name = ha_min_tier value = 5 }
				}
				else_if = {
					limit = { scope:rank_kingdom = yes }
					set_variable = { name = ha_min_tier value = 4 }
				}
				else_if = {
					limit = { scope:rank_duchy = yes }
					set_variable = { name = ha_min_tier value = 3 }
				}
				else_if = {
					limit = { scope:rank_county = yes }
					set_variable = { name = ha_min_tier value = 2 }
				}
			}
		}
	}

	send_option = { flag = expansionist_marry }
	send_option = { flag = rank_county }
	send_option = { flag = rank_duchy }
	send_option = { flag = rank_kingdom }
	send_option = { flag = rank_empire }
	send_option = { flag = rank_hegemon }
	send_options_exclusive = no
}

house_ambition_exempt = {
	is_shown = { scope:actor = { is_ai = no } }
	auto_accept = { always = yes }
	on_accept = {
		scope:recipient = {
			if = {
				limit = { NOT = { any_in_global_list = { variable = ha_exempt_list this = scope:recipient } } }
				add_to_global_variable_list = { name = ha_exempt_list target = scope:recipient }
				add_character_modifier = ha_exempt_modifier
			}
			else = {
				remove_list_global_variable = { name = ha_exempt_list target = scope:recipient }
				remove_character_modifier = ha_exempt_modifier
			}
		}
	}
}

# DEBUG: instantly run the matchmaker on the target, ignoring the toggle.
ha_test_marry = {
	is_shown = { scope:actor = { is_ai = no } }
	auto_accept = { always = yes }
	on_accept = {
		scope:recipient = {
			add_character_flag = ha_force
			trigger_event = house_ambition.1001
		}
	}
}
