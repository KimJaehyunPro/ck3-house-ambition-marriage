namespace = house_ambition

# Mirrors the proven-working backup architecture (single event, on_birthday).
# Heir filter uses direct relation checks (not a temp list inside a nested
# limit, which is unreliable in CK3).
house_ambition.1001 = {
	hidden = yes
	trigger = {
		OR = {
			any_in_global_list = { variable = ha_expansionist_list this = root.house }
			has_character_flag = ha_force
		}
		NOT = { any_in_global_list = { variable = ha_exempt_list this = root } }
		is_alive = yes
		is_adult = yes
		is_visibly_fertile = yes
		is_married = no
		is_betrothed = no
	}
	immediate = {
		remove_character_flag = ha_force
		save_scope_as = person_to_marry
		debug_log = "AMBITION: --- START MATCHMAKING ---"

		if = {
			limit = { house ?= { has_variable = ha_min_tier } }
			house = { save_scope_value_as = { name = ha_min value = var:ha_min_tier } }
		}
		else = {
			save_scope_value_as = { name = ha_min value = 2 }
		}

		if = {
			limit = { is_ruler = yes }
			save_scope_as = marriage_decider
		}
		else_if = {
			limit = { exists = employer }
			employer = { save_scope_as = marriage_decider }
		}
		else_if = {
			limit = { exists = liege }
			liege = { save_scope_as = marriage_decider }
		}

		if = {
			limit = { exists = scope:marriage_decider }

			scope:marriage_decider = {
				if = {
					limit = { exists = top_liege }
					top_liege = { save_scope_as = realm_anchor }
				}
				else = { save_scope_as = realm_anchor }
			}

			scope:realm_anchor = {
				add_to_temporary_list = local_rulers
				every_vassal_or_below = {
					limit = { highest_held_title_tier >= tier_county }
					add_to_temporary_list = local_rulers
				}

				if = {
					limit = { exists = capital_county }
					capital_county = {
						every_neighboring_county = {
							limit = { exists = holder }
							holder = {
								if = {
									limit = { exists = top_liege }
									top_liege = {
										add_to_temporary_list = local_rulers
										every_vassal_or_below = {
											limit = { highest_held_title_tier >= tier_county }
											add_to_temporary_list = local_rulers
										}
									}
								}
								else = {
									add_to_temporary_list = local_rulers
								}
							}
						}
					}
				}
			}

			every_in_list = {
				list = local_rulers

				limit = {
					trigger_if = {
						limit = { scope:ha_min >= 6 }
						is_independent_ruler = yes
						highest_held_title_tier >= tier_empire
					}
					trigger_else = {
						highest_held_title_tier >= scope:ha_min
					}
				}

				# CRITICAL: save the ruler we're harvesting from so the heir
				# filter can reach it via a concrete scope (temp list access
				# inside nested limits is unreliable in CK3).
				save_scope_as = source_ruler

				every_child = {
					save_scope_as = current_candidate
					limit = {
						NOT = { this = scope:person_to_marry }
						NOT = { highest_held_title_tier = tier_barony }
						is_adult = yes is_married = no is_betrothed = no is_imprisoned = no is_visibly_fertile = yes
						is_capable_adult = yes
						NOT = { has_trait = eunuch }
						NOT = { has_trait = devoted }
						NOT = { is_close_family_of = scope:person_to_marry }
						faith = { faith_hostility_level = { target = scope:person_to_marry.faith value < 2 } }
						OR = {
							AND = { scope:person_to_marry = { is_male = yes } is_female = yes }
							AND = { scope:person_to_marry = { is_female = yes } is_male = yes }
						}
						NOT = { dynasty = scope:person_to_marry.dynasty }
						NOT = { any_child = { always = yes } }

						could_marry_character_trigger = { CHARACTER = scope:person_to_marry }

						trigger_if = {
							limit = { scope:person_to_marry = { is_female = yes } }
							is_house_head = no
						}

						trigger_if = {
							limit = { exists = liege }
							liege = { opinion = { target = scope:person_to_marry value >= -20 } }
						}

						trigger_if = {
							limit = { scope:person_to_marry = { is_female = yes } }
							any_parent = { any_child = { count >= 2 is_male = yes } }
						}

						# HARD HEURISTIC: matrilineal candidate must NOT be the
						# eldest living son. Under feudal succession the eldest
						# son IS the primary heir and no AI lets you take their
						# primary heir your way. This catches the heir even if
						# the primary_heir character-link check below missed.
						trigger_if = {
							limit = { scope:person_to_marry = { is_female = yes } }
							any_sibling = {
								is_male = yes
								is_alive = yes
								age > prev.age
							}
						}

						# === HEIR FILTER (direct relation checks) ===
						# 1. The ruler we found this candidate through.
						NOT = { scope:source_ruler = { exists = primary_heir primary_heir = scope:current_candidate } }
						# 2. Any of the candidate's parents.
						NOT = { any_parent = { exists = primary_heir primary_heir = scope:current_candidate } }
						# 3. Candidate's liege.
						trigger_if = { limit = { exists = liege } NOT = { liege = { exists = primary_heir primary_heir = scope:current_candidate } } }
						# 4. Candidate's top liege.
						trigger_if = { limit = { exists = top_liege } NOT = { top_liege = { exists = primary_heir primary_heir = scope:current_candidate } } }
						# 5. Candidate's employer (for courtiers).
						trigger_if = { limit = { exists = employer } NOT = { employer = { exists = primary_heir primary_heir = scope:current_candidate } } }
					}
					add_to_temporary_list = valid_spouses
				}

				every_sibling = {
					save_scope_as = current_candidate
					limit = {
						NOT = { this = scope:person_to_marry }
						NOT = { highest_held_title_tier = tier_barony }
						is_adult = yes is_married = no is_betrothed = no is_imprisoned = no is_visibly_fertile = yes
						is_capable_adult = yes
						NOT = { has_trait = eunuch }
						NOT = { has_trait = devoted }
						NOT = { is_close_family_of = scope:person_to_marry }
						faith = { faith_hostility_level = { target = scope:person_to_marry.faith value < 2 } }
						OR = {
							AND = { scope:person_to_marry = { is_male = yes } is_female = yes }
							AND = { scope:person_to_marry = { is_female = yes } is_male = yes }
						}
						NOT = { dynasty = scope:person_to_marry.dynasty }
						NOT = { any_child = { always = yes } }

						could_marry_character_trigger = { CHARACTER = scope:person_to_marry }

						trigger_if = {
							limit = { scope:person_to_marry = { is_female = yes } }
							is_house_head = no
						}

						trigger_if = {
							limit = { exists = liege }
							liege = { opinion = { target = scope:person_to_marry value >= -20 } }
						}

						trigger_if = {
							limit = { scope:person_to_marry = { is_female = yes } }
							any_parent = { any_child = { count >= 2 is_male = yes } }
						}

						# HARD HEURISTIC: matrilineal candidate must NOT be the
						# eldest living son. Under feudal succession the eldest
						# son IS the primary heir and no AI lets you take their
						# primary heir your way. This catches the heir even if
						# the primary_heir character-link check below missed.
						trigger_if = {
							limit = { scope:person_to_marry = { is_female = yes } }
							any_sibling = {
								is_male = yes
								is_alive = yes
								age > prev.age
							}
						}

						# === HEIR FILTER (direct relation checks) ===
						# 1. The ruler we found this candidate through
						#    (their sibling who IS a ruler).
						NOT = { scope:source_ruler = { exists = primary_heir primary_heir = scope:current_candidate } }
						# 2. Any of the candidate's parents.
						NOT = { any_parent = { exists = primary_heir primary_heir = scope:current_candidate } }
						# 3. Candidate's liege.
						trigger_if = { limit = { exists = liege } NOT = { liege = { exists = primary_heir primary_heir = scope:current_candidate } } }
						# 4. Candidate's top liege.
						trigger_if = { limit = { exists = top_liege } NOT = { top_liege = { exists = primary_heir primary_heir = scope:current_candidate } } }
						# 5. Candidate's employer (for courtiers).
						trigger_if = { limit = { exists = employer } NOT = { employer = { exists = primary_heir primary_heir = scope:current_candidate } } }
					}
					add_to_temporary_list = valid_spouses
				}
			}

			random_in_list = {
				list = valid_spouses
				weight = {
					base = 10
					modifier = {
						add = 400
						exists = employer
						employer = {
							OR = {
								this = scope:marriage_decider.liege
								AND = { exists = liege liege = scope:marriage_decider.liege }
							}
						}
					}
					modifier = {
						add = 200
						exists = employer
						employer = {
							exists = top_liege
							top_liege = scope:realm_anchor
						}
					}
				}
				save_scope_as = best_candidate
			}
		}

		if = {
			limit = {
				exists = scope:best_candidate
				scope:best_candidate = {
					could_marry_character_trigger = { CHARACTER = scope:person_to_marry }
				}
			}
			debug_log = "AMBITION: SUCCESS - real match secured."
			scope:person_to_marry = {
				if = { limit = { is_female = yes } marry_matrilineal = scope:best_candidate }
				else = { marry = scope:best_candidate }
			}
		}
		else = {
			debug_log = "AMBITION: FAILSAFE - generating spouse."
			if = { limit = { NOT = { exists = scope:marriage_decider } } save_scope_as = marriage_decider }

			if = {
				limit = { scope:person_to_marry = { is_female = yes } }
				create_character = {
					random_traits = yes gender = male age = 16 employer = scope:marriage_decider
					dynasty = none faith = scope:person_to_marry.faith culture = scope:person_to_marry.culture
					save_scope_as = generated_spouse
				}
				scope:person_to_marry = { marry_matrilineal = scope:generated_spouse }
			}
			else = {
				create_character = {
					random_traits = yes gender = female age = 16 employer = scope:marriage_decider
					dynasty = none faith = scope:person_to_marry.faith culture = scope:person_to_marry.culture
					save_scope_as = generated_spouse
				}
				scope:person_to_marry = { marry = scope:generated_spouse }
			}
		}
	}
}
